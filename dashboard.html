async function loadRealProducts() { // On appelle ton serveur Render const response = await fetch('/api/products'); const data = await response.json(); const products = data.winners || []; // Tes 150 produits arrivent ici const container = document.getElementById('price-controls'); container.innerHTML = ''; // On vide // On groupe par niche (catégorie) const niches = ['energy', 'survival', 'water', 'tactical']; niches.forEach(niche => { const nicheProducts = products.filter(p => p.niche === niche); if(nicheProducts.length > 0) { container.innerHTML += `
AGENT ${niche.toUpperCase()} (${nicheProducts.length})
`; nicheProducts.forEach(p => { container.innerHTML += `
${p.name}
$
`; }); } }); }
